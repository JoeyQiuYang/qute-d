## Install

```bash
npm install --save qute-d
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'qute-d'
import 'qute-d/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
